    def self.create!(tag,description,indexed = true)
      path = tag.split('/')[0..-2].join('/')
      name = tag.split('/').last
      
      new_tag = new(:path => '/tags/'+path, :description => description, :indexed => indexed)
      new_tag.post!(:description => description, :name => name, :indexed => indexed)
    end
