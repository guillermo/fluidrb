    def method_missing(meth,*args)
      if meth.to_s =~ /=$/
        obj = Object.new(:path => self.path+"/"+ meth.to_s.gsub(/=$/,''))
        obj.update!(*args)
      else
        obj = (super || Object.new(:path => self.path+"/"+meth.to_s ))
      end
      obj
    end
